      const plugin = requirePlugin('quecPlugin')
      plugin.auto.autoSave({
        conditionType:1,
        icon:'',
        name:'自动化',
        nameType:2,
        conditions:[{                       // 必填。执行条件，最少1个，最多10个，满足所有条件时不能存在多个定时
                   "type": 1,                   // 必填。条件类型，0：设备，1：定时
                   "timer": {                   // 必填。定时数据
                       "type": 0,               // 必填。定时类型，0：仅一次，1：每天，2：自定义
                       "time": "17:10",           // 必填。定时执行时间点，符合HH:mm格式
                              "dayOfWeek": "",           // 非必填。重复周期，定时类型为自定义时必填，1,2,3...7表示周一到周日
                       "timeZone": "GMT+08:00"       // 必填。时区ID，符合GMT+-HH:mm格式
                   },
                   "sort": 1                    // 必填。条件顺序，从1开始递增
               },
               {
                   "type": 0,                            // 必填。条件类型，0：设备，1：定时
                   "productKey": "pe19SK",               // 必填。产品KEY，条件类型为设备时必填
                   "deviceKey": "VDU1187060484710334",   // 必填。设备KEY，条件类型为设备时必填
                   "property": {                         // 必填。设备属性
                       "id": 11,                 // 必填。物模型功能ID
                       "name": "int",                  // 必填。物模型名称
                       "code": "int",                  // 必填。物模型标识符
                       "value": 5,                     // 必填。物模型属性比较值
                       "dataType": "INT",              // 必填。数据类型，支持：BOOL、ENUM、INT、FLOAT、DOUBLE
                       "compare": ">",                 // 必填。比较符。>|>=|<|<=|==|!=。其中：BOOL、ENUM：!=、==，INT、FLOAT、DOUBLE：>、>=、<、<=、==、!=
                           "unit": ""                      // 非必填。单位
                   },
                   "sort": 2                             // 必填。条件顺序，从1开始递增
               }
           ],
        actions:[{                                     // 必填。执行动作，最少1个，最多50个，包含延时，延迟不能作为第一个动作
                   "type": 2,                     // 必填。执行动作类型，1：延时，2：设备，3：群组，4：场景
                   "productKey": "pe19SK",               // 必填。产品KEY
                   "deviceKey": "VDU1187060484710334",   // 必填。设备KEY
                        "name": "",                           // 必填。设备名称
                   "property": {                         // 必填。设备属性
                       "id": 12,                       // 必填。物模型功能ID
                       "name": "float",                // 必填。物模型名称
                       "code": "float",                // 必填。物模型标识符
                       "value": 1.1,                   // 必填。物模型属性值
                       "dataType": "INT",              // 必填。数据类型，支持：BOOL、ENUM、INT、FLOAT、DOUBLE
                        "subType": "RW",                // 非必填。读写权限
                            "unit": ""                      // 非必填。单位
                   },
                   "sort": 1                             // 必填。执行动作顺序，从1开始递增
               },
               {
                   "type": 1,                            // 必填。执行动作类型，1：延时，2：设备，3：群组，4：场景
                   "delayTime": 3,                  // 必填。延迟时长，单位秒，范围1-3599
                   "sort": 2                             // 必填。执行动作顺序，从1开始递增
               },
               {
                   "type": 3,                            // 必填。执行动作类型，1：延时，2：设备，3：群组，4：场景
                   "productKey": "pe19SK",               // 必填。群组默认PK
                   "deviceKey": "C1",                    // 必填。群组ID
                        "name": "",                           // 必填。群组名称
                   "property": {                         // 必填。设备属性
                       "id": 13,                       // 必填。物模型功能ID
                       "name": "double",               // 必填。物模型名称
                       "code": "double",               // 必填。物模型标识符
                       "value": 1.2,                   // 必填。物模型属性值
                       "dataType": "DOUBLE",           // 必填。数据类型，支持：BOOL、ENUM、INT、FLOAT、DOUBLE
                              "compare": ">",            // 必填。比较符。>|>=|<|<=|==|!=。其中：BOOL、ENUM：!=、==，INT、FLOAT、DOUBLE：>、>=、<、<=、==、!=
                         "subType": "RW",                // 非必填。读写权限
                             "unit": ""                      // 非必填。单位
                   },
                   "sort": 3                             // 必填。执行动作顺序，从1开始递增
               },
               {
                   "type": 4,                            // 必填。执行动作类型，1：延时，2：设备，3：群组，4：场景
                   "sceneId": "",                        // 必填。场景ID
                        "name": "",                           // 必填。场景名称
                   "sort": 4                             // 必填。执行动作顺序，从1开始递增
               }
           ],

            "effectDateType": 1,             // 必填。生效日期类型。0：每天，1：每周，2：每月，3：指定日期

               "effectTimeType": 3,             // 必填。生效时间类型。0：白天，1：夜晚，2：全天，3：指定时间段

               "timeZone": "GMT+08:00",      // 必填。时区ID，符合GMT+-HH:mm格式
               "location": "116.403,39.914"    // 非必填。经纬度，经度在前纬度在后，生效时间类型为白天或夜晚时必填

          接口调用失败的回调函数

        • success: (res: ResponseData) => void

          接口调用成功的回调函数

        • testResultId: String

          自动化id

      返回 void

      示例

      const plugin = requirePlugin('quecPlugin')
      plugin.auto.getTestAutomation({
        testResultId:xxx,
        success (res) {},
        fail (res) {},
        complete (res) {}
      })
